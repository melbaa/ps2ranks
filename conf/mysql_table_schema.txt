create database ps2ranks;
use ps2ranks;

CREATE TABLE `chartbl` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT,
  `rating` double NOT NULL,
  `rd` double NOT NULL,
  `volatility` double NOT NULL,
  `lastseen` int(10) unsigned NOT NULL,
  `lastopponent` varchar(1000) NOT NULL,
  `name` varchar(100) NOT NULL,
  `world` varchar(50) NOT NULL,
  `faction` varchar(50) NOT NULL,
  `br` bigint(10) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `name` (`name`),
  KEY `rating` (`rating`),
  KEY `world_rating` (`world`,`rating`)
) ENGINE=InnoDB AUTO_INCREMENT=7069306 DEFAULT CHARSET=ascii


CREATE TABLE `summarytbl` (
  `variable` varchar(100) NOT NULL,
  `value` varchar(100) NOT NULL,
  PRIMARY KEY (`variable`)
) ENGINE=MEMORY DEFAULT CHARSET=ascii
